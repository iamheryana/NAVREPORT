	UPDATE [dbo].[BC_ACCESS_FROM] SET [CODE] = 'BC'

	UPDATE [dbo].[BC_LAST_UPDATE] SET [Last Modified DateTime] = CAST('tgl GO LIVE YYYY-MM-DD' as datetime);
