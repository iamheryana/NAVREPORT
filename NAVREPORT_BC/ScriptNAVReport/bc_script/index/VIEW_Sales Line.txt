CREATE VIEW [dbo].[BC_AUTOJAYA$Sales Line$VSIFT$0] WITH SCHEMABINDING AS 
	SELECT "Document Type"
			,"Document No_"
			,"Line No_"
			,COUNT_BIG(*) AS "$Cnt"
			,SUM("Amount") AS "SUM$Amount"
			,SUM("Amount Including VAT") AS "SUM$Amount Including VAT"
	FROM dbo."BC_AUTOJAYA$Sales Line"
	GROUP BY "Document Type","Document No_","Line No_"
GO
